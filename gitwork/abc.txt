git init 创建库
git config --global user.name <name> 配置用户名称
git config --global user.email <email>配置用户邮箱
git touch <file name>创建文件
git rm <file name>删除文件
mkdir <file folder name>在当前目录创建文件夹
rm -r <file folder name>在当前目录删除文件夹
git add <file name>添加文件到本地版本库
git commit -m "描述的内容" 提交文件到本地版本库
git status 查询仓库当前状态
git push -u origin master
